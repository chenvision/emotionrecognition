flowchart TD

    C --> D1["Conv2D Kernel 3 (B, C1, T-2)"]
    C --> D2["Conv2D Kernel 4 (B, C2, T-3)"]
    C --> D3["Conv2D Kernel 5 (B, C3, T-4)"]
    D1 --> E1["ReLU + MaxPool (B, C1)"]
    D2 --> E2["ReLU + MaxPool (B, C2)"]
    D3 --> E3["ReLU + MaxPool (B, C3)"]
    E1 --> F["Concat Features"]
    E2 --> F
    E3 --> F
